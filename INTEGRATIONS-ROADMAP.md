```sql
-- New tables needed:

CREATE TABLE integrations (
  id UUID PRIMARY KEY,
  restaurant_id UUID REFERENCES restaurants(id),
  provider VARCHAR(50), -- 'planday', 'foodora', etc.
  status VARCHAR(20),   -- 'connected', 'error', 'disconnected'
  access_token TEXT,
  refresh_token TEXT,
  expires_at TIMESTAMP,
  settings JSONB,
  created_at TIMESTAMP,
  updated_at TIMESTAMP
);

CREATE TABLE sync_logs (
  id UUID PRIMARY KEY,
  integration_id UUID REFERENCES integrations(id),
  sync_type VARCHAR(50),  -- 'orders', 'revenue', 'staff', etc.
  status VARCHAR(20),      -- 'success', 'error'
  records_synced INTEGER,
  error_message TEXT,
  synced_at TIMESTAMP
);

CREATE TABLE external_orders (
  id UUID PRIMARY KEY,
  restaurant_id UUID,
  integration_id UUID REFERENCES integrations(id),
  external_id VARCHAR(255), -- Foodora/Wolt order ID
  platform VARCHAR(50),     -- 'foodora', 'wolt', 'direct'
  customer_name VARCHAR(255),
  items JSONB,
  total_amount DECIMAL,
  commission DECIMAL,
  net_revenue DECIMAL,
  status VARCHAR(50),
  created_at TIMESTAMP
);

CREATE TABLE social_posts (
  id UUID PRIMARY KEY,
  restaurant_id UUID,
  platform VARCHAR(50),   -- 'instagram', 'facebook'
  post_id VARCHAR(255),
  content TEXT,
  image_url TEXT,
  engagement JSONB,       -- likes, comments, shares
  posted_at TIMESTAMP
);
```
